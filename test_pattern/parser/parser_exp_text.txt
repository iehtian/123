result = (a != b);  // 不等于，结果为1
result = (a > b);   // 大于，结果为0
result = (a < b);   // 小于，结果为1
result = (a >= b);  // 大于等于，结果为0
result = (a <= b);  // 小于等于，结果为1

result = (c || d);  // 逻辑或，结果为1
result = !c;        // 逻辑非，结果为0

result = a ^ e;  // 位异或，结果为6（0110）
result = ~a;     // 位非，结果为-6（按位取反，二进制为...11111010）
result = a << 1; // 左移，结果为10（1010）
result = a >> 1; // 右移，结果为2（0010）
